All test categories live in sm-sdk/tests/ but:
  - Category 1: SDK language wrappers (4 tests per language - minimal)
  - Categories 2-4: Integration tests that orchestrate against external provider repos (Kubernetes, Terraform, Ansible)

  This avoids the "rebuild test infrastructure" problem by leveraging existing provider repos while centralizing coordination in sm-sdk.

  BaseIntegrationRunner.cs (Abstract base for Categories 2-4)

  using System;
  using System.Collections.Generic;
  using System.Diagnostics;
  using System.Linq;
  using System.Threading.Tasks;
  using SdkTestFramework.Models;

  namespace SdkTestFramework.Runners
  {
      public abstract class BaseIntegrationRunner
      {
          protected string IntegrationName { get; set; }
          protected string SdkRepoPath { get; set; }
          protected string AccessToken { get; set; }
          protected string OrganizationId { get; set; }
          protected string StateFile { get; set; }
          protected OperatingSystem[] SupportedOs { get; set; }

          public BaseIntegrationRunner(string integrationName, string sdkRepoPath, OperatingSystem[] supportedOs)
          {
              IntegrationName = integrationName;
              SdkRepoPath = sdkRepoPath;
              SupportedOs = supportedOs;

              AccessToken = Environment.GetEnvironmentVariable("ACCESS_TOKEN")
                  ?? throw new InvalidOperationException("ACCESS_TOKEN not set");
              OrganizationId = Environment.GetEnvironmentVariable("ORGANIZATION_ID")
                  ?? throw new InvalidOperationException("ORGANIZATION_ID not set");
              StateFile = Environment.GetEnvironmentVariable("STATE_FILE")
                  ?? throw new InvalidOperationException("STATE_FILE not set");
          }

          /// <summary>
          /// Runs integration tests for this category. Checks OS support first.
          /// </summary>
          public async Task<IntegrationTestResult> RunIntegrationTests()
          {
              var osContext = OsDetector.GetCurrentOsContext();
              var isSupported = OsDetector.IsPlatformSupported(SupportedOs);

              var result = new IntegrationTestResult
              {
                  IntegrationName = IntegrationName,
                  Os = osContext.OsDisplayName,
                  Architecture = osContext.Architecture,
                  Supported = isSupported
              };

              if (!isSupported)
              {
                  result.SkipReason = $"Not supported on {osContext.OsDisplayName}";
                  return result;
              }

              var stopwatch = Stopwatch.StartNew();

              try
              {
                  var operations = await ExecuteIntegrationTests();
                  result.Operations.AddRange(operations);
              }
              catch (Exception ex)
              {
                  result.Operations.Add(new TestOperation
                  {
                      Name = "setup",
                      Success = false,
                      Error = ex.Message
                  });
              }

              stopwatch.Stop();
              result.TotalDurationMs = stopwatch.ElapsedMilliseconds;
              result.Passed = result.Operations.Count(o => o.Success);
              result.Failed = result.Operations.Count(o => !o.Success);

              return result;
          }

          protected abstract Task<List<TestOperation>> ExecuteIntegrationTests();

          protected async Task<TestOperation> RunIntegrationTest(string testName, string scriptPath)
          {
              var env = new Dictionary<string, string>
              {
                  { "TEST_NAME", testName },
                  { "ACCESS_TOKEN", AccessToken },
                  { "ORGANIZATION_ID", OrganizationId },
                  { "STATE_FILE", StateFile }
              };
